aule.metrics.climate

Climate-science specific metrics, operating along the time axis (T).

These metrics require the temporal dimension to be meaningful (i.e. input shapes (batch, H, W, C, T) or (H, W, C, T)); for inputs without a time axis, T defaults to 1 and these metrics degrade gracefully but are less informative.

 21def seasonal_error(
 22    y_true: np.ndarray,
 23    y_pred: np.ndarray,
 24    data_format: Optional[str] = None,
 25    ignore_nan: bool = False,
 26) -> float:
 27    '''
 28        Computes the MSE between the spatial-mean time series of ground
 29        truth and prediction (i.e. how well the model preserves the
 30        temporal/seasonal trend, regardless of spatial detail).
 31
 32        Parameters:
 33        -----------
 34        - y_true : np.ndarray
 35            Ground truth array, any of the 4 supported shapes.
 36        - y_pred : np.ndarray
 37            Prediction array, same shape as y_true.
 38        - data_format : str
 39            "bhwc" or "hwct", required only when arrays are 4D.
 40        - ignore_nan : bool
 41            If True, non-finite values are excluded when computing spatial
 42            means (default: False).
 43
 44        Returns:
 45        --------
 46        - value : float
 47            MSE between spatial-mean time series. Lower is better.
 48
 49        Usage:
 50        ------
 51
 52        ```python
 53        import numpy as np
 54        from aule.metrics import seasonal_error
 55
 56        gt   = np.random.rand(64, 64, 1, 365)
 57        pred = gt + np.random.normal(0, 0.05, gt.shape)
 58        score = seasonal_error(gt, pred, data_format="hwct")
 59        ```
 60    '''
 61
 62    y_true_c, y_pred_c = match_shapes(y_true, y_pred, data_format=data_format)
 63
 64    if ignore_nan:
 65        true_mean = np.nanmean(np.where(np.isfinite(y_true_c), y_true_c, np.nan), axis=(1, 2))
 66        pred_mean = np.nanmean(np.where(np.isfinite(y_pred_c), y_pred_c, np.nan), axis=(1, 2))
 67    else:
 68        true_mean = np.mean(y_true_c, axis=(1, 2))
 69        pred_mean = np.mean(y_pred_c, axis=(1, 2))
 70
 71    return float(np.mean((true_mean - pred_mean) ** 2))

 74def percentile_error(
 75    y_true: np.ndarray,
 76    y_pred: np.ndarray,
 77    percentile: float = 95.0,
 78    data_format: Optional[str] = None,
 79    ignore_nan: bool = False,
 80) -> float:
 81    '''
 82        Computes the absolute error between a given percentile of the ground
 83        truth distribution and the same percentile of the prediction
 84        distribution. Useful for assessing how well a model captures
 85        extreme values (e.g. P95, P99 for heatwaves; P5, P1 for cold extremes).
 86
 87        Parameters:
 88        -----------
 89        - y_true : np.ndarray
 90            Ground truth array, any of the 4 supported shapes.
 91        - y_pred : np.ndarray
 92            Prediction array, same shape as y_true.
 93        - percentile : float
 94            Percentile to compare, in [0, 100] (default: 95.0).
 95        - data_format : str
 96            "bhwc" or "hwct", required only when arrays are 4D.
 97        - ignore_nan : bool
 98            If True, non-finite values are excluded from the percentile
 99            computation (default: False).
100
101        Returns:
102        --------
103        - value : float
104            Absolute difference between the two percentile values. Lower is better.
105
106        Usage:
107        ------
108
109        ```python
110        import numpy as np
111        from aule.metrics import percentile_error
112
113        gt   = np.random.rand(8, 64, 64, 1)
114        pred = gt * 0.9
115        score = percentile_error(gt, pred, percentile=95.0)
116        ```
117    '''
118
119    y_true_c, y_pred_c = match_shapes(y_true, y_pred, data_format=data_format)
120
121    true_flat = y_true_c.astype(np.float64).ravel()
122    pred_flat = y_pred_c.astype(np.float64).ravel()
123
124    if ignore_nan:
125        true_flat = true_flat[np.isfinite(true_flat)]
126        pred_flat = pred_flat[np.isfinite(pred_flat)]
127
128    true_p = np.percentile(true_flat, percentile)
129    pred_p = np.percentile(pred_flat, percentile)
130
131    return float(np.abs(true_p - pred_p))

134def pixelwise_temporal_correlation(
135    y_true: np.ndarray,
136    y_pred: np.ndarray,
137    data_format: Optional[str] = None,
138    ignore_nan: bool = False,
139) -> np.ndarray:
140    '''
141        Computes the Pearson correlation coefficient at each pixel, across
142        the batch/time dimension, between ground truth and prediction.
143        Requires a meaningful sample dimension (batch or time) to correlate over.
144
145        Parameters:
146        -----------
147        - y_true : np.ndarray
148            Ground truth array, any of the 4 supported shapes.
149        - y_pred : np.ndarray
150            Prediction array, same shape as y_true.
151        - data_format : str
152            "bhwc" or "hwct", required only when arrays are 4D.
153        - ignore_nan : bool
154            If True, non-finite values are excluded from the per-pixel
155            statistics (default: False).
156
157        Returns:
158        --------
159        - r_map : np.ndarray
160            Array of shape (H, W, C) with the Pearson r at each pixel,
161            averaged over the time axis if both batch and time are present.
162
163        Usage:
164        ------
165
166        ```python
167        import numpy as np
168        from aule.metrics import pixelwise_temporal_correlation
169
170        gt   = np.random.rand(100, 64, 64, 1)
171        pred = gt + np.random.normal(0, 0.1, gt.shape)
172        r_map = pixelwise_temporal_correlation(gt, pred)  # shape (64, 64, 1)
173        ```
174    '''
175
176    y_true_c, y_pred_c = match_shapes(y_true, y_pred, data_format=data_format)
177    B, H, W, C, T = y_true_c.shape
178
179    # Merge batch and time into a single "sample" axis (N, H, W, C) to correlate over.
180    # Move the (B, T) axes to the front, then flatten them together.
181    true_samples = np.moveaxis(y_true_c, 4, 1).reshape(B * T, H, W, C)
182    pred_samples = np.moveaxis(y_pred_c, 4, 1).reshape(B * T, H, W, C)
183
184    n = true_samples.shape[0]
185    if n < 2:
186        raise ValueError(
187            "pixelwise_temporal_correlation requires at least 2 samples "
188            "along batch or time; got a single sample."
189        )
190
191    if ignore_nan:
192        valid = finite_mask(true_samples, pred_samples)
193        true_samples = np.where(valid, true_samples, np.nan)
194        pred_samples = np.where(valid, pred_samples, np.nan)
195        mean_fn, sum_fn = np.nanmean, np.nansum
196    else:
197        mean_fn, sum_fn = np.mean, np.sum
198
199    true_mean = mean_fn(true_samples, axis=0)
200    pred_mean = mean_fn(pred_samples, axis=0)
201
202    true_centered = true_samples - true_mean
203    pred_centered = pred_samples - pred_mean
204
205    numerator = sum_fn(true_centered * pred_centered, axis=0)
206    true_var = sum_fn(true_centered ** 2, axis=0)
207    pred_var = sum_fn(pred_centered ** 2, axis=0)
208    denominator = np.sqrt(np.clip(true_var * pred_var, a_min=1e-12, a_max=None))
209
210    r_map = np.clip(numerator / denominator, -1.0, 1.0)
211
212    return r_map

215def trend_error(
216    y_true: np.ndarray,
217    y_pred: np.ndarray,
218    data_format: Optional[str] = None,
219    ignore_nan: bool = False,
220) -> float:
221    '''
222        Computes the absolute difference between the linear trend (slope of
223        a least-squares fit over time) of the ground truth spatial-mean time
224        series and that of the prediction. Useful for checking whether a
225        model reproduces long-term warming/cooling or drying/wetting trends,
226        independent of how well it matches short-term variability.
227
228        Parameters:
229        -----------
230        - y_true : np.ndarray
231            Ground truth array, with a time axis (shapes (b) or (d)).
232        - y_pred : np.ndarray
233            Prediction array, same shape as y_true.
234        - data_format : str
235            "bhwc" or "hwct", required only when arrays are 4D. Must be
236            "hwct" here, since a time axis is required.
237        - ignore_nan : bool
238            If True, non-finite values are excluded from the spatial means (default: False).
239
240        Returns:
241        --------
242        - value : float
243            Absolute difference between the two trend slopes (units per time step).
244
245        Usage:
246        ------
247
248        ```python
249        import numpy as np
250        from aule.metrics import trend_error
251
252        t = np.arange(100)
253        gt   = np.tile((0.01 * t).reshape(1, 1, 1, -1), (32, 32, 1, 1))
254        pred = gt + np.random.normal(0, 0.05, gt.shape)
255        score = trend_error(gt, pred, data_format="hwct")
256        ```
257    '''
258
259    y_true_c = to_canonical(y_true, data_format=data_format)
260    y_pred_c = to_canonical(y_pred, data_format=data_format)
261
262    if y_true_c.shape[-1] < 2:
263        raise ValueError(
264            "trend_error requires at least 2 time steps; "
265            f"got T={y_true_c.shape[-1]}. Did you forget data_format='hwct'?"
266        )
267
268    if ignore_nan:
269        true_mean = np.nanmean(np.where(np.isfinite(y_true_c), y_true_c, np.nan), axis=(0, 1, 2, 3))
270        pred_mean = np.nanmean(np.where(np.isfinite(y_pred_c), y_pred_c, np.nan), axis=(0, 1, 2, 3))
271    else:
272        true_mean = np.mean(y_true_c, axis=(0, 1, 2, 3))
273        pred_mean = np.mean(y_pred_c, axis=(0, 1, 2, 3))
274
275    t = np.arange(len(true_mean), dtype=np.float64)
276
277    true_slope = float(np.polyfit(t, true_mean, 1)[0])
278    pred_slope = float(np.polyfit(t, pred_mean, 1)[0])
279
280    return float(np.abs(true_slope - pred_slope))

283def _event_durations(series: np.ndarray, threshold: float, above: bool) -> np.ndarray:
284    '''
285        Finds the durations (in time steps) of consecutive runs where a
286        1D series is above (or below) a threshold.
287
288        Parameters:
289        -----------
290        - series : np.ndarray
291            1D time series.
292        - threshold : float
293            Threshold value.
294        - above : bool
295            If True, finds runs where series > threshold; otherwise series < threshold.
296
297        Returns:
298        --------
299        - durations : np.ndarray
300            1D array of run lengths (in time steps). Empty if no event occurs.
301    '''
302
303    mask = series > threshold if above else series < threshold
304    if not np.any(mask):
305        return np.array([], dtype=np.int64)
306
307    padded = np.concatenate(([0], mask.astype(np.int64), [0]))
308    diffs = np.diff(padded)
309    starts = np.where(diffs == 1)[0]
310    ends = np.where(diffs == -1)[0]
311
312    return (ends - starts).astype(np.int64)

315def extreme_event_duration_error(
316    y_true: np.ndarray,
317    y_pred: np.ndarray,
318    threshold: float,
319    above: bool = True,
320    data_format: Optional[str] = None,
321    ignore_nan: bool = False,
322) -> float:
323    '''
324        Computes the absolute difference in mean extreme-event duration
325        (e.g. heatwave or cold-wave length) between ground truth and
326        prediction, based on the spatial-mean time series crossing a
327        fixed threshold. An "event" is a run of consecutive time steps
328        above (or below) the threshold.
329
330        Parameters:
331        -----------
332        - y_true : np.ndarray
333            Ground truth array, with a time axis (shapes (b) or (d)).
334        - y_pred : np.ndarray
335            Prediction array, same shape as y_true.
336        - threshold : float
337            Threshold value defining an extreme event (e.g. a heatwave temperature).
338        - above : bool
339            If True (default), an event is a run above the threshold
340            (e.g. heatwave); if False, a run below it (e.g. cold wave).
341        - data_format : str
342            "bhwc" or "hwct", required only when arrays are 4D. Must be
343            "hwct" here, since a time axis is required.
344        - ignore_nan : bool
345            If True, non-finite values are excluded from the spatial means (default: False).
346
347        Returns:
348        --------
349        - value : float
350            Absolute difference in mean event duration (in time steps).
351            Returns 0.0 if neither series has any qualifying event.
352
353        Usage:
354        ------
355
356        ```python
357        import numpy as np
358        from aule.metrics import extreme_event_duration_error
359
360        gt   = np.random.rand(32, 32, 1, 60) + 0.5
361        pred = gt + np.random.normal(0, 0.05, gt.shape)
362        score = extreme_event_duration_error(gt, pred, threshold=0.8, data_format="hwct")
363        ```
364    '''
365
366    y_true_c = to_canonical(y_true, data_format=data_format)
367    y_pred_c = to_canonical(y_pred, data_format=data_format)
368
369    if ignore_nan:
370        true_mean = np.nanmean(np.where(np.isfinite(y_true_c), y_true_c, np.nan), axis=(0, 1, 2, 3))
371        pred_mean = np.nanmean(np.where(np.isfinite(y_pred_c), y_pred_c, np.nan), axis=(0, 1, 2, 3))
372    else:
373        true_mean = np.mean(y_true_c, axis=(0, 1, 2, 3))
374        pred_mean = np.mean(y_pred_c, axis=(0, 1, 2, 3))
375
376    true_durations = _event_durations(true_mean, threshold, above)
377    pred_durations = _event_durations(pred_mean, threshold, above)
378
379    true_avg = float(np.mean(true_durations)) if true_durations.size > 0 else 0.0
380    pred_avg = float(np.mean(pred_durations)) if pred_durations.size > 0 else 0.0
381
382    return float(np.abs(true_avg - pred_avg))

385def autocorrelation_error(
386    y_true: np.ndarray,
387    y_pred: np.ndarray,
388    max_lag: int = 10,
389    data_format: Optional[str] = None,
390    ignore_nan: bool = False,
391) -> float:
392    '''
393        Computes the mean absolute error between the temporal autocorrelation
394        function (ACF) of the ground truth spatial-mean time series and that
395        of the prediction, up to a maximum lag. Useful for checking whether
396        a model preserves temporal persistence/memory (e.g. drought
397        persistence, slow climate modes).
398
399        Parameters:
400        -----------
401        - y_true : np.ndarray
402            Ground truth array, with a time axis (shapes (b) or (d)).
403        - y_pred : np.ndarray
404            Prediction array, same shape as y_true.
405        - max_lag : int
406            Maximum lag (in time steps) to compare (default: 10).
407        - data_format : str
408            "bhwc" or "hwct", required only when arrays are 4D. Must be
409            "hwct" here, since a time axis is required.
410        - ignore_nan : bool
411            If True, non-finite values are excluded from the spatial means (default: False).
412
413        Returns:
414        --------
415        - value : float
416            Mean absolute error between the two ACFs over lags 1..max_lag. Lower is better.
417
418        Usage:
419        ------
420
421        ```python
422        import numpy as np
423        from aule.metrics import autocorrelation_error
424
425        gt   = np.cumsum(np.random.randn(32, 32, 1, 200), axis=-1)
426        pred = gt + np.random.normal(0, 0.1, gt.shape)
427        score = autocorrelation_error(gt, pred, max_lag=10, data_format="hwct")
428        ```
429    '''
430
431    y_true_c = to_canonical(y_true, data_format=data_format)
432    y_pred_c = to_canonical(y_pred, data_format=data_format)
433
434    T = y_true_c.shape[-1]
435    if T <= max_lag:
436        raise ValueError(
437            f"autocorrelation_error requires more time steps than max_lag; "
438            f"got T={T}, max_lag={max_lag}."
439        )
440
441    if ignore_nan:
442        true_mean = np.nanmean(np.where(np.isfinite(y_true_c), y_true_c, np.nan), axis=(0, 1, 2, 3))
443        pred_mean = np.nanmean(np.where(np.isfinite(y_pred_c), y_pred_c, np.nan), axis=(0, 1, 2, 3))
444    else:
445        true_mean = np.mean(y_true_c, axis=(0, 1, 2, 3))
446        pred_mean = np.mean(y_pred_c, axis=(0, 1, 2, 3))
447
448    def _acf(series: np.ndarray, max_lag: int) -> np.ndarray:
449        series = series - np.mean(series)
450        var = np.dot(series, series)
451        if var == 0:
452            return np.zeros(max_lag)
453        return np.array([np.dot(series[:-lag], series[lag:]) / var for lag in range(1, max_lag + 1)])
454
455    true_acf = _acf(true_mean, max_lag)
456    pred_acf = _acf(pred_mean, max_lag)
457
458    return float(np.mean(np.abs(true_acf - pred_acf)))
